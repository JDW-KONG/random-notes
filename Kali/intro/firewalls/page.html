- ufw is an interface for iptables
- apt-get install ufw
- ufw allow 22/tcp
- ufw deny 80/tcp
- ufw deny 2000:3000/tcp
- ufw allow from 192.168.1.1 to any port 443 proto tcp
- ufw enable
- ufw status
- ufw logging enabled
- ufw status verbose
- ufw status numbered
- ufw delete [rule number]
- ufw deny in on eth0 from 192.168.1.1/24
- ufw reset
- resets rules
